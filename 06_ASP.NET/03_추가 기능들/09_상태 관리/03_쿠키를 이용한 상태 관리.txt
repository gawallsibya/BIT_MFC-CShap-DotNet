<script runat="server">
    
    int tempCount = 0;
    
    protected void Page_Load(object sender, EventArgs e)
    {
        if (Request.Cookies["Counter"] == null)
        {
            Response.Cookies["Counter"].Value = "0";
            DisplayCounter();
        }
    }

    void DisplayCounter()
    {
        Label1.Text = tempCount.ToString();
    }


    protected void Button1_Click(object sender, EventArgs e)
    {
        tempCount = int.Parse(Request.Cookies["Counter"].Value) + 1;
        Response.Cookies["Counter"].Value = tempCount.ToString();
        Response.Cookies["Counter"].Expires = DateTime.Now.AddSeconds(10);
        DisplayCounter();
    }

    protected void Button2_Click(object sender, EventArgs e)
    {
        tempCount = int.Parse(Request.Cookies["Counter"].Value) - 1;
        Response.Cookies["Counter"].Value = tempCount.ToString();
        Response.Cookies["Counter"].Expires = DateTime.Now.AddSeconds(10);
        DisplayCounter();
    }
</script>

<html xmlns="http://www.w3.org/1999/xhtml" >
<head id="Head1" runat="server">
    <title>제목 없음</title>
</head>
<body>
    <form id="form1" runat="server">
    <div>
        <h3>쿠키를 이용한 카운터 예제</h3>
        
        <asp:Label ID="Label1" runat="server"></asp:Label>
        <asp:Button ID="Button1" runat="server" Text="+" OnClick="Button1_Click" />
        <asp:Button ID="Button2" runat="server" Text="-" OnClick="Button2_Click" />
    </div>
    </form>
</body>
</html>
